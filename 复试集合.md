# 质数筛

```c
const int N = 1e6 + 10;
int primes[N];
bool st[N];
int cnt;
void di(int t) {
    for (int i = 2; i <= t; i ++ ) {
        if (st[i]) continue;
        primes[cnt ++ ] = i;
        for (int j = 2 * i; j <= t; j += i)
            st[j] = 1;
    }
}
```

# 快速幂和逆元

```c
LL qmi(LL a, int b, int p) {
    LL ans = 1;
    while (b) {
        if (b & 1) ans = ans * a % p;
        a = a * a % p;
        b >>= 1;
    }
    return ans;
}
qmi(a, p - 2, p);
a ^ a = 0
```
